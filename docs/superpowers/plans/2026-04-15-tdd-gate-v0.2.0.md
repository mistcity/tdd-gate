```typescript
function findProjectRoot(startDir: string): string | null {
  let dir = startDir;
  for (let i = 0; i < 10; i++) {
    if (
      existsSync(path.join(dir, '.git')) ||
      existsSync(path.join(dir, 'tdd-gate.config.json')) ||
      existsSync(path.join(dir, 'package.json'))
    ) {
      return dir;
    }
    const parent = path.dirname(dir);
    if (parent === dir) return null;
    dir = parent;
  }
  return null;
}
```

```typescript
import { execFileSync } from 'node:child_process';
import path from 'node:path';
import type { TddGateConfig, ImpactResult } from '../types.js';
import { classifyFile, LANGUAGES } from './file-classifier.js';
import type { Journal } from './journal.js';

const LANGUAGE_BY_EXT: Record<string, (typeof LANGUAGES)[number]> = {};
for (const lang of LANGUAGES) {
  for (const ext of lang.extensions) {
    LANGUAGE_BY_EXT[ext] = lang;
  }
}

export function buildImportPattern(basename: string, language: string): string {
  switch (language) {
    case 'typescript':
    case 'javascript':
    case 'tsx':
    case 'jsx':
      return `(?:from\\s+['"].*\\b${basename}['"]|require\\(\\s*['"].*\\b${basename}['"]\\s*\\))`;
    case 'python':
      return `(?:from\\s+\\.?${basename}\\s+import|import\\s+.*\\b${basename}\\b)`;
    case 'go':
      return `["'].*/${basename}["']`;
    case 'kotlin':
    case 'java':
      return `import\\s+.*\\.${basename}\\b`;
    case 'rust':
      return `(?:use\\s+.*::${basename}\\b|mod\\s+${basename}\\b)`;
    case 'csharp':
      return `using\\s+.*\\.${basename}\\b`;
    default:
      return basename;
  }
}
```

```typescript
interface FindDependentsOptions {
  maxFiles: number;
  timeout: number;
}

export function findDependents(
  filePath: string,
  language: string,
  projectRoot: string,
  options: FindDependentsOptions,
): string[] {
  const basename = path.basename(filePath, path.extname(filePath));
  const pattern = buildImportPattern(basename, language);
  const ext = path.extname(filePath);

  try {
    const result = execFileSync('grep', [
      '-rl', '--include', `*${ext}`, '-E', pattern, projectRoot,
    ], {
      timeout: options.timeout,
      encoding: 'utf-8',
      stdio: ['pipe', 'pipe', 'pipe'],
    }).trim();

    if (!result) return [];

    return result.split('\n')
      .slice(0, options.maxFiles)
      .filter((f) => {
        if (path.resolve(f) === path.resolve(filePath)) return false;
        const fileBasename = path.basename(f, path.extname(f));
        const lang = LANGUAGE_BY_EXT[ext];
        if (lang && lang.isTestFile(fileBasename, ext)) return false;
        return true;
      });
  } catch {
    return [];
  }
}
```

```typescript
  projectRoot: string,
  config: TddGateConfig,
  journal: Journal,
): ImpactResult[] {
  if (!config.impactAnalysis) return [];
  if (journal.hasTestRun()) return [];

  const results: ImpactResult[] = [];
  const options = {
    maxFiles: config.impactAnalysisMaxFiles,
    timeout: config.impactAnalysisTimeout,
  };

  for (const filePath of changedImplFiles) {
    const ext = path.extname(filePath);
    const lang = LANGUAGE_BY_EXT[ext];
    if (!lang) continue;

    const dependents = findDependents(filePath, lang.name, projectRoot, options);
    if (dependents.length === 0) continue;

    const uncoveredDependents: string[] = [];
    const missingTests: string[] = [];

    for (const dep of dependents) {
      const depClassification = classifyFile(dep, config);
      if (depClassification.type !== 'impl') continue;
      if (!journal.hasTestFor(depClassification.expectedTests)) {
        uncoveredDependents.push(dep);
        missingTests.push(depClassification.expectedTests[0] || path.basename(dep));
      }
    }

    if (uncoveredDependents.length > 0) {
      results.push({ filePath, dependents: uncoveredDependents, missingTests });
    }
  }

  return results;
}
```
